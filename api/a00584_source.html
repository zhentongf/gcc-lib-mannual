366  auto __output = [&](size_t __idx)

390  continue;

392  if (*__fmt_first == '&')

394  __output(0);
395  continue;

398  }

400  *__out++ = '\\';
401  }
402  else

417  return true;

419  return false;
420  };
421

423  *__out++ = '$';
424  else if (__eat('$'))

427  __output(0);
